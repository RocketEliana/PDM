override fun onCreate(savedInstanceState: Bundle?) {
    super.onCreate(savedInstanceState)
    enableEdgeToEdge()

    // 1. PRIMERO: Creamos el binding (Construimos la casa)
    binding = ActivityMainBinding.inflate(layoutInflater)
    
    // 2. SEGUNDO: Le decimos a la actividad que use esta vista
    setContentView(binding.root)

    // 3. TERCERO: Ahora ya podemos tocar los elementos (el Toolbar ya existe)
    setSupportActionBar(binding.toolbar)
    supportActionBar?.title = "Lista de Unicornios"

    // 4. CUARTO: La lÃ³gica del fragmento
    if (savedInstanceState == null) {
        supportFragmentManager.beginTransaction()
            .setReorderingAllowed(true)
            .replace(binding.fragmentContainerView.id, FragmentoSpinner())
            .commit()
    }

    // 5. QUINTO: Ajustes de pantalla completa
    ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
        val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
        v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
        insets
    }
}----------
