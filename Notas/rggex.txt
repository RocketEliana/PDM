fun esDNIValido(dni: String): Boolean {
    // Regex para comprobar formato: 8 d√≠gitos + letra
    val regex = Regex("^\\d{8}[A-HJ-NP-TV-Z]\$", RegexOption.IGNORE_CASE)
    if (!regex.matches(dni)) return false

    // Letras v√°lidas seg√∫n el n√∫mero
    val letras = "TRWAGMYFPDXBNJZSQVHLCKE"

    // N√∫mero y letra
    val numero = dni.substring(0, 8).toInt()
    val letra = dni[8].uppercaseChar()

    // Comprobar que la letra coincide
    return letra == letras[numero % 23]
}
