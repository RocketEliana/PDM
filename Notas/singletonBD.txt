package com.example.contactos

import android.content.Context
import androidx.room.Database
import androidx.room.Room
import androidx.room.RoomDatabase

@Database(entities=[Contacto::class],version=1)

abstract class AppDataBase : RoomDatabase(){
    abstract fun ContactoDAO() : ContactoDAO
    companion object {
        @Volatile
        private var INSTANCE: AppDataBase? = null

        fun getDataBase(context: Context): AppDataBase {
            return INSTANCE ?: synchronized(this) {
                val instance = Room.databaseBuilder(
                    context.applicationContext,
                    AppDataBase::class.java,
                    "contactos_db"
                ).build()
                INSTANCE = instance
                instance
            }
        }


    }

}
/*Explicación rápida

@Volatile → asegura que si varias hebras acceden a INSTANCE, vean siempre la versión más reciente.

synchronized(this) → evita que dos hilos creen la base de datos a la vez.

Room.databaseBuilder(...) → crea la base con:

contexto de aplicación (no activity),

la clase AppDataBase::class.java,

el nombre "contactos_db".*/
